FileOperation constructor. Checks for incorrect invokation (i.e., without new keyword). Options must be specified and may be {}. options.root is not necessary, but is strongly recommended.

 
var DEFAULT_ROOT = '/';
var FileOperation = function(options) {
    if (!(this instanceof FileOperation)) {
        return new FileOperation(options);
    }
    if (!options) {
        throw new TypeError("First argument of FileInformation must be object or string");
    }
    
    var self = this;
    
    var root = (typeof options === 'object')
                ? options.root || DEFAULT_ROOT
                : (typeof options === 'string')
                    ? options
                    : DEFAULT_ROOT;

    self.root = function() {
        return root;
    };
